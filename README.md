function shuffle(arr) {
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  return arr;
}

const el = (id) => document.getElementById(id);

const state = {
  all: DATA,
  quiz: [],
  idx: 0,
  answers: [], // selected option index or null
  correct: [], // boolean or null if unanswered
};

function setView(which) {
  el("setup").style.display = (which==="setup") ? "block" : "none";
  el("quiz").style.display = (which==="quiz") ? "block" : "none";
  el("result").style.display = (which==="result") ? "block" : "none";
}

function startQuiz(count) {
  const all = [...state.all];
  shuffle(all);
  const n = Math.max(1, Math.min(count, all.length));
  state.quiz = all.slice(0, n);
  state.idx = 0;
  state.answers = new Array(n).fill(null);
  state.correct = new Array(n).fill(null);
  el("kSel").textContent = String(n);
  setView("quiz");
  render();
}

function resetAll() {
  state.quiz = [];
  state.idx = 0;
  state.answers = [];
  state.correct = [];
  el("kSel").textContent = "0";
  el("reviewWrap").style.display = "none";
  setView("setup");
}

function answeredCount() {
  return state.answers.filter(v => v !== null).length;
}

function scoreCount() {
  let s = 0;
  for (let i=0;i<state.correct.length;i++) if (state.correct[i] === true) s++;
  return s;
}

function render() {
  const q = state.quiz[state.idx];
  const n = state.quiz.length;
  el("posTag").textContent = `Q ${state.idx+1}/${n}`;
  el("lessonTag").textContent = q.lesson || "Lezione";
  el("pageTag").textContent = `Pag. ${q.page}`;
  el("qtext").textContent = q.question;

  const answered = answeredCount();
  const score = scoreCount();
  el("answered").textContent = String(answered);
  el("score").textContent = String(score);
  el("bar").style.width = `${Math.round(((state.idx+1)/n)*100)}%`;

  const container = el("answers");
  container.innerHTML = "";
  const selected = state.answers[state.idx];

  q.options.forEach((opt, i) => {
    const btn = document.createElement("button");
    btn.className = "ans";
    btn.type = "button";
    btn.textContent = opt;

    const locked = (selected !== null);
    if (locked) {
      btn.classList.add("locked");
      if (i === q.correct) btn.classList.add("correct");
      if (i === selected && selected !== q.correct) btn.classList.add("wrong");
      btn.disabled = true;
    } else {
      btn.addEventListener("click", () => choose(i));
    }
    container.appendChild(btn);
  });

  el("prevBtn").disabled = (state.idx === 0);
  el("nextBtn").disabled = (state.idx === n-1);
}

function choose(i) {
  const q = state.quiz[state.idx];
  state.answers[state.idx] = i;
  state.correct[state.idx] = (i === q.correct);
  render();

  if (i === q.correct) {
    el("hint").innerHTML = `✅ <b>Giusta!</b>`;
  } else {
    el("hint").innerHTML = `❌ <b>Sbagliata.</b> La risposta giusta era evidenziata nel PDF.`;
  }
}

function go(delta) {
  const n = state.quiz.length;
  state.idx = Math.max(0, Math.min(n-1, state.idx + delta));
  el("hint").textContent = "Clicca una sola risposta per registrare la scelta.";
  render();
}

function finish() {
  const n = state.quiz.length;
  const s = scoreCount();
  const perc = n ? Math.round((s/n)*100) : 0;

  el("finalScore").textContent = `${s} / ${n}`;
  el("finalPerc").textContent = `${perc}%`;

  setView("result");
  el("reviewWrap").style.display = "none";
}

function buildReview() {
  const list = el("reviewList");
  list.innerHTML = "";
  state.quiz.forEach((q, i) => {
    const div = document.createElement("div");
    div.className = "review-item";

    const ok = (state.correct[i] === true);
    const dot = `<span class="dot ${ok ? "ok" : "bad"}"></span>`;
    const sel = state.answers[i];
    const selTxt = (sel === null) ? "—" : q.options[sel];
    const corTxt = q.options[q.correct];

    div.innerHTML = `
      <div class="r-q">${dot}<b>${i+1}.</b> ${escapeHtml(q.question)} <span class="mono">(${escapeHtml(q.lesson || "")} • Pag. ${q.page})</span></div>
      <div class="r-a"><b>Hai scelto:</b> ${escapeHtml(selTxt)}</div>
      <div class="r-a"><b>Corretta:</b> ${escapeHtml(corTxt)}</div>
    `;
    list.appendChild(div);
  });
}

function escapeHtml(s) {
  return String(s)
    .replaceAll("&","&amp;")
    .replaceAll("<","&lt;")
    .replaceAll(">","&gt;");
}
